Les types de variables

Il existe plusieurs types de variables

Par défaut, le type d'une variable est 'NULL'

La première fonction pour connaître le type est 'typeof'


var foo_bar = 'lorem ipsum';

console.log(typeof(foo_bar)); // "string
console.log(typeof foo_bar);  // "string

Les types de variables


typeof 10;            // "number"
typeof 10.5;          // "number"
typeof 'lorem ipsum'; // "string"
typeof function(){};  // "function"
typeof true;          // "boolean"

typeof {};            // "object"
typeof [];            // "object"
typeof null;          // "object"
typeof /abcd/;        // "object"
typeof document.createElement('div'); // 'object'

typeof ne fonctionne pas avec certaines variables

Les types de variables

La solution


function type(o)
{
    var s = Object.prototype.toString.call(o);
    return s.match(/\[object (.*?)\]/)[1].toLowerCase();
}

type({});           // "object"
type([]);           // "array"
type(5);            // "number"
type(null);         // "null"
type();             // "undefined"
type(/abcd/);       // "regex"
type(new Date());   // "date"
type(function(){}); // "function"

Les types de variables

Une version plus complète qui gère les cas particuliers

À mettre dans tous vos projets


function type(o)
{
    if (o === null)
        return 'null';

    if (o && (o.nodeType === 1 || o.nodeType === 9))
        return 'element';

    var s    = Object.prototype.toString.call(o),
        type = s.match(/\[object (.*?)\]/)[1].toLowerCase();

    if (type === 'number')
    {
        if (isNaN(o))
            return 'nan';
        if (!isFinite(o))
            return 'infinity';
    }

    return type;
}

Les variables 'globales'

console.log(window.var_1);

Si une variable est déclarée en dehors de toute fonction, elle est qualifiée de globale.

Elle sera ajoutée à l'objet window

Le scope

Une variable peut être surchargée par les paramètres d'une fonction si elle a le même nom


var var_1 = 1;

function my_function(var_1)
{
    console.log(var_1);
}

my_function(2);
console.log(var_1);

Le résultat sera "2" suivi de "1"

Dans la fonction, var_1 correspond au paramètre mais en dehors de la fonction, var_1 correspond à la variable globale
